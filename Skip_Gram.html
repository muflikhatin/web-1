import torch
import torch.nn as nn
import torch.optim as optim
from collections import Counter
import numpy as np

# Misalnya, kita memiliki kamus kata-kata
kata = set(vectorizer.vocabulary_.keys())

# Menghitung jumlah kata dalam kamus
vocab_size = len(kata)

# Membuat kamus untuk mengonversi kata menjadi indeks (dan sebaliknya)
vocab_to_int = {word: idx for idx, word in enumerate(kata)}
int_to_vocab = {idx: word for idx, word in enumerate(kata)}

# Fungsi untuk membuat representasi one-hot dari indeks kata
def one_hot_encoding(word_idx, vocab_size):
    one_hot = torch.zeros(vocab_size)
    one_hot[word_idx] = 1
    return one_hot

# Menginisialisasi model Skip-gram dengan representasi one-hot
class SkipGram(nn.Module):
    def __init__(self, vocab_size, embedding_dim):
        super(SkipGram, self).__init__()
        self.embeddings = nn.Embedding(vocab_size, embedding_dim)
        self.linear = nn.Linear(embedding_dim, vocab_size)

    def forward(self, target):
        target_embeds = self.embeddings(target)
        output = self.linear(target_embeds)
        return output

# Membuat instance model Skip-gram
embedding_dim = 100
model = SkipGram(vocab_size, embedding_dim)

# Inisialisasi list untuk menyimpan representasi one-hot untuk setiap kata
one_hot_vectors = []

for word in kata:
    # Mengonversi kata menjadi indeks
    word_idx = vocab_to_int.get(word, -1)

    if word_idx != -1:
        # Membuat representasi one-hot dari indeks kata
        one_hot_vector = one_hot_encoding(word_idx, vocab_size)
        one_hot_vectors.append(one_hot_vector)
    else:
        print(f"'{word}' tidak ada dalam kamus.")

# one_hot_vectors sekarang berisi representasi one-hot untuk semua kata dalam variabel kata

# Untuk melihat struktur model Anda, Anda dapat mencetaknya dengan perintah berikut:
print("\nStruktur Model Skip-gram:")

# Fungsi untuk training model
def train(model, data, vocab_to_int, epochs=5, lr=0.001):
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(model.parameters(), lr=lr)

    for epoch in range(epochs):
        total_loss = 0

        for sentence in data:
            # Memisahkan kata-kata dalam kalimat
            words = sentence.split()

            for target_word in words:
                # Memeriksa apakah kata ada dalam vocabulary
                if target_word in vocab_to_int:
                    target = torch.LongTensor([vocab_to_int[target_word]])
                    optimizer.zero_grad()
                    output = model(target)
                    loss = criterion(output, target)
                    loss.backward()
                    optimizer.step()
                    total_loss += loss.item()

        print(f'Epoch {epoch+1}/{epochs}, Loss: {total_loss}')

# Training model
data = dt['cleaning']
train(model, data, vocab_to_int)
